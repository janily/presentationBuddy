焊接第一个元件：Hello Agent

// [01] Import Core
import { Agent } from '@mastra/core';

// [02] Instantiate Agent
export const agent = new Agent({
  name: 'HELLO_AGENT',
  model: 'gpt-4o', // [03] Bind Model
  instructions: 'You are a helpful assistant.'
});

// [04] Run via Local Dev Server
await agent.generate('Hello Mastra!');

接通工具引脚与本地 Playground

TOOL NODE: weather.ts
import { createTool } from '@mastra/core';
import { z } from 'zod';

export const weatherTool = createTool({
  id: 'get_weather',
  schema: z.object({
    city: z.string()
  }),
  execute: async ({city}) => {
    return `Weather in ${city}: Sunny`;
  }
});
