MCP 代码演示 (1)：定义 MCP Server 与 Schema

import { Server } from '@modelcontextprotocol/sdk/server/index.js'; import { z } from 'zod'; const server = new Server({ name: 'weather-mcp', version: '1.0.0' }); server.setRequestHandler('tools/list', () => ({ tools: [{ name: 'weatherTool', inputShape: z.object({ city: z.string() }) }] })); server.setRequestHandler('tools/call', async (req) => { const { city } = req.params; return { temperature: 22, city }; });

综合示例：MCP + Skill 协同工作

import { Agent, McpTool, defineSkill } from '@mastra/core'; const mcp = new McpTool({ serverUrl: 'http://localhost:3000/mcp', toolName: 'weatherTool' }); const skill = defineSkill({ name: 'local', params: z.object({ x: z.number() }), async run({ x }) { return x * 2; } }); const agent = new Agent({ name: 'hybrid', tools: [mcp, skill], model: 'gpt-4o' }); const out = await agent.generate( 'Get weather via MCP and double temp via localSkill'); console.log(out.text);
